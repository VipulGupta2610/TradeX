function generateCandles(base, n = 500, volatility = 0.015) {
  const out = [];
  let p = base;
  const now = Date.now();
  for (let i = n; i >= 0; i--) {
    const drift = (Math.random() - 0.48) * p * volatility;
    const open = p;
    const close = Math.max(open + drift, open * 0.92);
    const range = Math.abs(drift) + Math.random() * p * 0.006;
    const high = Math.max(open, close) + Math.random() * range;
    const low = Math.min(open, close) - Math.random() * range;
    const volume = 50000 + Math.random() * 2000000;
    out.push({ open, high, low, close, volume, time: now - i * 60000 });
    p = close;
  }
  return out;
}

const fmt = (p, dec=2) => {
  if (p == null || isNaN(p)) return "0.00";
  const n = Number(p);
  if (n > 100000) return n.toLocaleString("en-IN", { maximumFractionDigits: dec });
  return n.toFixed(dec);
};
const fmtVol = (v) => {
  if (v >= 1e7) return (v/1e7).toFixed(2)+"Cr";
  if (v >= 1e5) return (v/1e5).toFixed(2)+"L";
  if (v >= 1000) return (v/1000).toFixed(1)+"K";
  return v.toFixed(0);
};

// ── MARKDOWN RENDERER ─────────────────────────────────────────────
function Markdown({ text }) {
  if (!text) return null;
  return (
    <div style={{ fontSize:12.5, lineHeight:1.75, color:"#c5cee0" }}>
      {text.split("\n").map((line,i) => {
        if (line.startsWith("### ")) return <div key={i} style={{fontWeight:700,color:"#e8edf5",fontSize:13,marginTop:10,marginBottom:3}}>{line.slice(4)}</div>;
        if (line.startsWith("## ")) return <div key={i} style={{fontWeight:800,color:"#f0f4fb",fontSize:14,marginTop:12,marginBottom:4,borderBottom:"1px solid rgba(255,255,255,0.08)",paddingBottom:5}}>{line.slice(3)}</div>;
        if (line.startsWith("| ")) return <div key={i} style={{fontFamily:"'JetBrains Mono',monospace",fontSize:11,color:"#64748b",borderBottom:"1px solid rgba(255,255,255,0.04)",padding:"2px 0"}}>{line}</div>;
        if (line === "---") return <hr key={i} style={{border:"none",borderTop:"1px solid rgba(255,255,255,0.08)",margin:"8px 0"}}/>;
        if (line === "") return <div key={i} style={{height:6}}/>;
        const parts = line.split(/(\*\*.*?\*\*|`.*?`)/g);
        const isBullet = line.startsWith("• ") || line.startsWith("- ");
        const content = (
          <span>{parts.map((p,j) => {
            if (p.startsWith("**") && p.endsWith("**")) return <strong key={j} style={{color:"#e2e8f0",fontWeight:700}}>{p.slice(2,-2)}</strong>;
            if (p.startsWith("`") && p.endsWith("`")) return <code key={j} style={{fontFamily:"'JetBrains Mono',monospace",background:"rgba(255,255,255,0.09)",padding:"1px 5px",borderRadius:4,fontSize:11}}>{p.slice(1,-1)}</code>;
            return <span key={j}>{p}</span>;
          })}</span>
        );
        if (isBullet) return <div key={i} style={{paddingLeft:16,position:"relative",color:"#8fa3c4",marginBottom:3}}><span style={{position:"absolute",left:3,color:"#3b82f6"}}>›</span>{content}</div>;
        return <div key={i} style={{color:"#8fa3c4",marginBottom:2}}>{content}</div>;
      })}
    </div>
  );
}

function Toast({ message, type, onDone }) {

  const C = { success:"#089981", error:"#f23645", info:"#3b82f6", warning:"#f59e0b" };
  const c = C[type] || C.info;
  const icons = { success:"✓", error:"✕", info:"ℹ", warning:"⚠" };
  return (
    <div style={{padding:"12px 18px",background:"#181d2e",border:`1px solid ${c}33`,borderLeft:`3px solid ${c}`,borderRadius:10,boxShadow:"0 8px 32px rgba(0,0,0,0.6)",display:"flex",alignItems:"center",gap:12,animation:"toastIn 0.3s cubic-bezier(0.34,1.56,0.64,1)",minWidth:300,maxWidth:420}}>

            </div>

        {/* ─── AI PANEL ─────────────────────────────────────────── */}
        {showAI && (
          <div className="fade-up" style={{width:340,background:"#0d1117",borderLeft:"1px solid rgba(59,130,246,0.18)",display:"flex",flexDirection:"column",flexShrink:0,overflow:"hidden"}}>

              {aiMessages.map((m,i)=>(
                <div key={i} className={m.role==="user"?"ai-msg-user":"ai-msg-bot"}>
                  {m.role==="assistant"&&(
                    <div style={{display:"flex",alignItems:"center",gap:6,marginBottom:8}}>
                      <div style={{width:18,height:18,borderRadius:5,background:"linear-gradient(135deg,#3b82f6,#7c3aed)",display:"flex",alignItems:"center",justifyContent:"center",fontSize:9,color:"#fff"}}>✦</div>

                    <div style={{width:18,height:18,borderRadius:5,background:"linear-gradient(135deg,#3b82f6,#7c3aed)",display:"flex",alignItems:"center",justifyContent:"center",fontSize:9}}>✦</div>

              <div style={{position:"relative"}}>

                  placeholder={`Ask about ${activeSym.sym}…`} rows={2}
                  style={{width:"100%",paddingRight:42,background:"#111827",border:"1px solid rgba(59,130,246,0.18)",borderRadius:10,padding:"10px 42px 10px 14px",fontSize:12.5,color:"#e2e8f0",resize:"none",outline:"none",lineHeight:1.5,transition:"border-color 0.15s"}}/>

      {/* ═══ DIALOGS ══════════════════════════════════════════════ */}
      {showOrderDialog && (
        <OrderDialog sym={activeSym.sym} price={showOrderDialog.price} side={showOrderDialog.side}
          onClose={()=>setShowOrderDialog(null)} onConfirm={handleOrder}/>
      )}
      {showAlertDialog && (
        <AlertDialog sym={activeSym.sym} price={activePrice}
          onClose={()=>setShowAlertDialog(false)}
          onSet={(a)=>{ setAlerts(p=>[...p,a]); addToast(`Alert set for ${a.sym} @ ₹${fmt(a.price)}`,"warning"); }}/>
      )}

      {/* ═══ TOASTS ═══════════════════════════════════════════════ */}
      <div style={{position:"fixed",bottom:24,right:24,zIndex:2000,display:"flex",flexDirection:"column",gap:8,alignItems:"flex-end"}}>
        {toasts.map(t=>(

        ))}
      </div>
    </div>
  );
